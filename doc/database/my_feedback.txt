CREATE DATABASE `my-feedback` DEFAULT CHARSET utf8 COLLATE utf8_general_ci;

USE `web`;

CREATE TABLE `fb_user` (
  `fb_user_id` int(10) unsigned NOT NULL AUTO_INCREMENT,
  `fb_user_card_id` varchar(18) DEFAULT NULL,
  `fb_user_name` varchar(10) DEFAULT NULL,
  `fb_user_card_photo` varchar(30) DEFAULT NULL,
  `fb_user_phone` varchar(11) DEFAULT NULL,
  `fb_user_status` tinyint(4) DEFAULT NULL,
  `fb_user_password` varchar(32) DEFAULT NULL,
  `create_at` timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP,
  `update_at` timestamp NULL DEFAULT NULL,
  PRIMARY KEY (`fb_user_id`)
) ENGINE=InnoDB AUTO_INCREMENT=2 DEFAULT CHARSET=utf8;

DELIMITER //
CREATE TRIGGER fb_user_update_trigger BEFORE UPDATE ON fb_user
FOR EACH ROW
BEGIN
 SET NEW.update_at = NOW();
END;
//
DELIMITER ;

DELIMITER //
CREATE TRIGGER fb_user_insert_trigger BEFORE INSERT ON fb_user
FOR EACH ROW
BEGIN
 SET NEW.update_at = NOW();
END;
//
DELIMITER ;
